
SELECT COD_CONTRATO, DATA, total, 'UNION 1' AS QUERY
FROM   tcontrato
WHERE  TOTAL >= 1000 AND TOTAL < 2000

UNION  ALL

SELECT COD_CONTRATO, DATA, total, 'UNION 2' AS QUERY
FROM   tcontrato
WHERE  desconto IS NOT NULL

UNION

SELECT COD_CONTRATO, DATA, total, 'UNION 3' AS QUERY
FROM   tcontrato
WHERE  total > 2000

ORDER  BY QUERY;
