CREATE OR REPLACE PROCEDURE exclui_contrato
  (pCod IN NUMBER, pMsg OUT VARCHAR)
AS
BEGIN
  DELETE FROM TCONTRATO
  WHERE COD_CONTRATO = pCod;
  pMsg := SQL%ROWCOUNT ||' Contrato Excluido ';
END;
