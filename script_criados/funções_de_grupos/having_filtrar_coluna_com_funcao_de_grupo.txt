--Having - filtrar coluna com funcao de grupo
SELECT COD_ALUNO, AVG(TOTAL) MEDIA
FROM TCONTRATO
WHERE COD_ALUNO > 0
HAVING AVG(Total) > 500
GROUP BY COD_ALUNO
ORDER BY COD_ALUNO

--
SELECT Trunc(DATA), SUM(TOTAL) TOTAL
FROM   TCONTRATO
WHERE  COD_CONTRATO > 0
GROUP  BY Trunc(DATA)
HAVING SUM(TOTAL) > 500
ORDER  BY TOTAL Desc;
