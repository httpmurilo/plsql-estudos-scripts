DECLARE
  CURSOR c1 IS
    SELECT * FROM TAluno;
  Reg TAluno%ROWTYPE;
BEGIN
  FOR reg IN c1  --open, la√ßo, fetch, close, exit when
  LOOP
    Dbms_Output.Put_Line('Codigo: '||
       LPad(reg.cod_aluno,5,'0')||' - ' || 'Nome: '||reg.nome);
  END LOOP;
END;
